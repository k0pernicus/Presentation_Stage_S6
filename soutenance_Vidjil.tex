	\subsection{Contexte biologique}
	\begin{frame}
	\begin{block}{Généralités sur l'ADN}
		\pause
		\textbf{ADN}: molécule modélisée en double-hélice, support universel de l'information génétique.
		\\
		Constituée d'un alphabet de 4 lettres: \textbf{A},\textbf{T},\textbf{G},\textbf{C}.
		\\
		Peut être sujette à des \textbf{recombinaisons}, par différents phénomènes biologiques.
	\end{block}
	\pause
	\begin{block}{Les Leucocytes: gardiens de notre corps}
		\pause
		\textbf{Leucocyte = globule blanc.}
		\\
		Les \textbf{lymphocytes} font partie d'une classe de leucocytes, présents dans le sang.
		\\
		Un lymphocyte reconnaît un agent étranger \textbf{spécifique} en fonction de son \textbf{site actif} (\textbf{recombinaison V(D)J}).
		\\
		Lors de la reconnaissance: activation -> multiplication.
	\end{block}
	\end{frame}
	
	\begin{frame}
		\vfill
		Chaque lymphocyte porte une portion d'ADN V(D)J permettant son identification.
		\vfill
		\begin{block}{Recombinaison V(D)J}
		\begin{enumerate}
			\item{Délétion de portions du gène V et J}
			\item{Ajout de nucléotides aléatoires entre V/D et D/J}
		\end{enumerate}
		\end{block}
	\end{frame}
	
	\begin{frame}
	\begin{block}{La Leucémie Aigüe Lymphoblastique (LAL)}
		\pause
		Cancer liquide affectant majoritairement les enfants.
		\\
		Provoque une \textbf{surproduction de lymphocytes cancéreux}. Les lymphocytes strictement identiques, regroupés en population, sont appelés \textbf{clones V(D)J}.
	\end{block}
	\pause
	\begin{alertblock}{Problème}
		\pause
		L'accumulation de lymphocytes cancéreux provoque:
		\begin{itemize}
			\item{une anémie chez le patient,}
			\item{une diminution du nombre de plaquettes,}
			\item{une diminution des globules blancs normaux.}
		\end{itemize}
		Peut amener à la mort, chez le malade.
	\end{alertblock}
	\end{frame}

	\section{A la recherche de la distance d'édition perdue...}
	
	\begin{frame}
	\begin{block}{Qu'est-ce qu'une distance d'édition ?}
		\pause
		\textbf{\textit{Une distance d'édition est un nombre de modifications à apporter selon trois opérations, permettant de passer d'une chaîne de caractères à une autre.}}
		\\
		3 opérations: insertion/délétion/substitution.
		\\
		Nous prendrons le poids suivant: \textbf{0} pour un \textit{match}, \textbf{1} pour insertion/délétion/substitution -> \textit{Levenshtein}.
	\end{block}
	\pause
	\begin{exampleblock}{Exemple 1}
		Prenons deux chaînes de caractères de même longueur: CHIEN / NICHE.
		\\
		\begin{tabular}{llllll}
		    & C & H & I & E & N \\
		N & - & - & - & - & - \\
		I & - & - & - & - & - \\
		C & - & - & - & - & - \\
		H & - & - & - & - & - \\
		E & - & - & - & - & - \\
		\end{tabular}
	\end{exampleblock}
	\end{frame}
	
	\begin{frame}
	\begin{block}{Qu'est-ce qu'une distance d'édition ?}
		\pause
		\textbf{\textit{Une distance d'édition est un nombre de modifications à apporter selon trois opérations, permettant de passer d'une chaîne de caractères à une autre.}}
		\\
		3 opérations: insertion/délétion/substitution.
		\\
		Nous prendrons le poids suivant: \textbf{0} pour un \textit{match}, \textbf{1} pour insertion/délétion/substitution -> \textit{Levenshtein}.
	\end{block}
	\begin{exampleblock}{Exemple 1}
		Prenons deux chaînes de caractères de même longueur: CHIEN / NICHE.
		\\
		\begin{tabular}{llllll}
		    & C & H & I & E & N \\
		N & \textbf{1} & 2 & 3 & 4 & 4 \\
		I & \textbf{2} & 2 & 2 & 3 & 4 \\
		C & \textbf{2} & 3 & 3 & 3 & 4 \\
		H & 3& \textbf{2} & \textbf{3} & 4 & 4 \\
		E & 4 & 3 & 3 & \textbf{3} & \textbf{4} \\
		\end{tabular}
	\end{exampleblock}
	\end{frame}
	
	\begin{frame}
	\begin{block}{Qu'est-ce qu'une distance d'édition ?}
		\pause
		\textbf{\textit{Une distance d'édition est un nombre de modifications minimal à apporter selon trois opérations, permettant de passer d'une chaîne de caractères à une autre.}}
		\\
		3 opérations: insertion/délétion/substitution.
		\\
		Nous prendrons le poids suivant: \textbf{0} pour un \textit{match}, \textbf{1} pour insertion/délétion/substitution -> \textit{Levenshtein}.
	\end{block}
	\begin{exampleblock}{Exemple 1}
		Prenons deux chaînes de caractères de même longueur: CHIEN / NICHE.
		\\
		Le résultat du calcul est: \textbf{--CH-E-}:
		\begin{itemize}
		\item{\textbf{deux insertions}: N, I,}
		\item{\textbf{deux délétions}: I, N.}
		\end{itemize}
		Soit une distance d'édition de \textbf{4}.
	\end{exampleblock}
	\end{frame}
	
	\begin{frame}
	\vfill
		Implantation du calcul dans le programme C++.
		\\
		Environ 5000 distances retournées pour 100 clones, reprises dans l'\textit{interface}
	\pause
	\vfill
	\begin{block}{Mais, que calcule-t-on?}
		\pause
		Le calcul des distances d'édition se fait par comparaison sur deux \textbf{séquences de 40 nucléotides} comprises entre la fin de la région V, et le début de la région J = les \textbf{\textit{windows}}
	\end{block}
	\pause
	\vfill
	\begin{alertblock}{Pourquoi les \textit{windows?}}
		\pause
		Window = marqueur spécifique d'un lymphocyte.
		\\
		Avantage par rapport au suivi au cours du temps, en calcul de distances,...
	\end{alertblock}
	\end{frame}

	\subsection{Le graphe de distances d'édition}
	\begin{frame}
		\begin{block}{Qu'est-ce qu'un graphe?}
			\pause
			En informatique, un graphe est une structure de données représentée par deux éléments de base:
			\begin{itemize}
			\item{les noeuds,}
			\item{les arêtes.}
			\end{itemize}
		\end{block}
		\pause
		\begin{block}{Idée de la visualisation}
				\begin{itemize}
					\item{Noeuds: clones,}
					\item{Arêtes: distances.}
				\end{itemize}
			\begin{itemize}
			\item{Visualisation des distances en fonction d'une distance d'édition maximale, dans le \textit{scatterPlot}.}
			\\
			\item{Permission de faire varier la distance d'édition maximale, de façon interactive (avec l'utilisation d'un \textit{slider}).}
			\\
			\item{Implémentation en HTML5, Javascript et les frameworks D3JS et JQuery.}
			\end{itemize}
		\end{block}
	\end{frame}

		Deux idées:
		\begin{itemize}
			\item{visualisation en un seul graphe,}

	\subsection{Le graphe DBSCAN}
	
	\begin{frame}
	
		\vfill
	
		Visualisation basée sur l'\textbf{algorithme de clusterisation DBSCAN}
		
		\pause
		
		\vfill
		
		\begin{block}{DBSCAN}
			DBSCAN = Density-Based Spatial Clustering of Applications with Noise
			\\ 
			Algorithme publié en 1996 par \textbf{Martin Ester}, \textbf{Hans-Peter Kriegel}, \textbf{Jörg Sander} et \textbf{Xiaowei Xu}.
			\\
			Utilise la densité estimée des clusters (la distance d'édition - $\epsilon$ - ainsi que le nombre de voisin minimum - \textit{MinPts} - d'un clone) pour partitionner.
		\end{block}
		
	\end{frame}
	
	\begin{frame}
	
	\begin{exampleblock}{Exemple}
	
	\begin{center}
	
		\includegraphics[scale=0.5]{img/DBSCAN_Wiki_ex.png}
		\\
		Exemple DBSCAN à \textit{MinPts} = 0 - \textbf{en.wikipedia.org}
	
	\end{center}
	
	\end{exampleblock}
	
	\end{frame}
	
	\begin{frame}
		\vfill
		\begin{block}{Idée de la visualisation}
		\begin{itemize}
			\item{Permission de faire varier $\epsilon$ ainsi que \textit{MinPts} de façon interactive pour partitionner.}
			\item{Instanciation d'un objet DBSCAN à chaque modification d'un/des paramètre(s).}
			\item{Cluster = sphère contenant le voisinage, autour du noeud \textit{CORE}.}
			\item{Ajout d'une couleur aléatoire pour chaque cluster, pour mieux les différencier sur cette visualisation et les précédentes.}
			\item{HTML5, Javascript, et les frameworks D3JS et JQuery - création d'une classe objet spécifique pour faire de l'algorithme un objet + tests unitaires QUnit.}
		\end{itemize}
		\end{block}
	\end{frame}
